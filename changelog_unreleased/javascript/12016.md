#### Add jsxSpaceBeforeSelfClosingTag option (#12016 by @sryze)

Make it possible to customize whether to put a space before `/>` in self-closing tags in JSX. On by default.

<!-- prettier-ignore -->
```jsx
// Input
function Component() {
  return (
    <div>
      <img/>
    </div>
  );
}

// Prettier stable
function Component() {
  return (
    <div>
      <img />
    </div>
  );
}

// Prettier main
function Component() {
  return (
    <div>
      <img/>
    </div>
  );
}
```
